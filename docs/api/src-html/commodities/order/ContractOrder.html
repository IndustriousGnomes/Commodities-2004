287            } else {
288                this.date = date;
289            }
290

292            this.month              = month;
293            this.commodityName      = commodityName;
294            this.desiredPrice       = desiredPrice;
295            this.stopPrice          = stopPrice;

585        /**
586         *  Set the status for this order.
587         *  @param  status  the order's status
588         */
589        public  void setStatus(String status) {
590            if (!this.status.equals(status)) {
591                this.status = status;
592                CommodityCalendar calendar = (CommodityCalendar)CommodityCalendar.getInstance();
593                calendar.clearTime();
594    
595                if (ContractPanel.NO_STATUS.equals(status)) {
596                    delete();
597                } else if (ContractPanel.CONSIDERATION.equals(status)) {
598                    dateConsidered = calendar.getTime();
599                    datePlaced = null;
600                    offsettingOrder = -1;
601                    recordOrder();
602                } else if (ContractPanel.PLACED.equals(status)) {
603                    datePlaced = calendar.getTime();
604                    dateFilled = null;
605                    offsettingOrder = -1;
606                    recordOrder();
607                } else if (ContractPanel.FILLED.equals(status)) {
608                    dateFilled = calendar.getTime();
609                    dateCancelled = null;
610                    dateStopLoss = null;
611                    recordOrder();
612                } else if (ContractPanel.CANCEL_PLACED.equals(status)) {
613                    dateCancelPlaced = calendar.getTime();
614                    dateFilled = null;
615                    dateCancelled = null;
616                    offsettingOrder = -1;
617                    recordOrder();
618                } else if (ContractPanel.CANCELLED.equals(status)) {
619                    dateCancelled = calendar.getTime();
620                    dateFilled = null;
621                    offsettingOrder = -1;
622                    recordOrder();
623                } else if (ContractPanel.STOP_LOSS.equals(status)) {
624                    dateStopLoss = calendar.getTime();
625                    recordOrder();
626                }
627            }
628        }

657        /**
658         *  Indicates if this order is equal to another order.  Equal, in this
659         *  case, indicates that the orders have the same account number, order type,
660         *  initial date, contract, desired price, stoploss price, status, and actual price.
661         *  It does not include the number of contracts.
662         *
663         *  @param  toOrder     the order to compare this one to
664         */
665        public boolean equals(ContractOrder toOrder) {
666            if (acctNumber == toOrder.getAccountNumber() &&
667                orderType == toOrder.isLongOrder() &&
668                date.equals(toOrder.getDate()) &&
669                month.equals(toOrder.getMonth()) &&
670                commodityName.equals(toOrder.getCommodityName()) &&
671                desiredPrice.equals(toOrder.getDesiredPrice()) &&
672                stopPrice.equals(toOrder.getStopPrice()) &&
673                actualPrice.equals(toOrder.getActualPrice()) &&
674                status.equals(toOrder.getStatus())) {
675                return true;
676            } else {
677                return false;
678            }
679        }

681        /**
682         *  Creates a new order with the requested number of contracts
683         *  in the new order and deducted from the original order.  If
684         *  the number of contracts in the new order is not less than
685         *  the original order, then a null order is returned and the
686         *  original order remains unchanged.
687         *
688         *  @param  numberInNew     number of contracts in new order
689         */
690        public ContractOrder split(int numberInNew) {
691            ContractOrder order = null;
692            if (numberInNew >= number.intValue()) {
693                return null;
694            }
695            number = new Integer(number.intValue() - numberInNew);
696            recordOrder();
697            order = new ContractOrder(-1,
698                                      acctNumber,
699                                      orderType,
700                                      date,
701                                      new Integer(numberInNew),
702                                      month,
703                                      commodityName,
704                                      desiredPrice,
705                                      stopPrice,
706                                      status,
707                                      actualPrice,
708                                      note,
709                                      -1,
710                                      dateConsidered,
711                                      datePlaced,
712                                      dateFilled,
713                                      dateCancelPlaced,
714                                      dateCancelled,
715                                      dateStopLoss);
716            return order;
717        }

719        /**
720         *  Combines the addOrder with this order.  All dates fields are
721         *  updated with the most current date between the two orders.
722         *  The addOrder is also deleted.
723         *
724         *  @param  addOrder    the order to be added into this order
725         *  @return             true if the combine was successful
726         */
727        public boolean combine(ContractOrder addOrder) {
728            if ((addOrder == null) ||
729                !this.equals(addOrder) ||
730                (addOrder.getNumberOfContracts().intValue() <= 0)) {
731                return false;
732            }
733    
734            number = new Integer(number.intValue() + addOrder.getNumberOfContracts().intValue());
735    
736            dateConsidered  = (dateConsidered == null || dateConsidered.compareTo(addOrder.getDateConsidered()) < 0)?addOrder.getDateConsidered():dateConsidered;
737            datePlaced      = (datePlaced== null || datePlaced.compareTo(addOrder.getDatePlaced()) < 0)?addOrder.getDatePlaced():datePlaced;
738            dateFilled      = (dateFilled == null || dateFilled.compareTo(addOrder.getDateFilled()) < 0)?addOrder.getDateFilled():dateFilled;
739            dateCancelPlaced = (dateCancelPlaced == null || dateCancelPlaced.compareTo(addOrder.getDateCancelPlaced()) < 0)?addOrder.getDateCancelPlaced():dateCancelPlaced;
740            dateCancelled   = (dateCancelled == null || dateCancelled.compareTo(addOrder.getDateCancelled()) < 0)?addOrder.getDateCancelled():dateCancelled;
741            dateStopLoss    = (dateStopLoss == null || dateStopLoss.compareTo(addOrder.getDateStopLoss()) < 0)?addOrder.getDateStopLoss():dateStopLoss;
742    
743            recordOrder();
744            addOrder.delete();
745            return true;
746        }
