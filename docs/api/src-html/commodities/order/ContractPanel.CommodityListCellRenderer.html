382            /**
383             *  Consolidate the orders in the table based on the passed in order and the
384             *  row that it currently resides in.  The order will be the order to be consolidated
385             *  into a row above its current location.
386             *
387             *  @param  order   the order to consolidate on
388             *  @param  row     the row the order is originally on
389             */
390            private void consolidateOrders(ContractOrder order, int row) {
391    
392                ContractOrderLine   orderLine = (ContractOrderLine)rowData.get(row);
393                ContractOrderLine   line = null;
394                ContractOrder       origOrder = order;
395                ContractOrder       checkOrder = null;
396                ContractOrder       offsettingOrder = null;
397                int                 orderIndex = STATUSES.indexOf(origOrder.getStatus());
398                if (orderIndex == -1) orderIndex = 0;
399                int                 checkOrderIndex = 0;
400                int                 checkTilRow = row;
401                boolean             firstSwap = true;
402    
403                for (int i = 0; i < checkTilRow; i++) {
404                    line = (ContractOrderLine)rowData.get(i);
405                    if (origOrder.isLongOrder()) {
406                        checkOrder = line.getLongOrder();
407                        offsettingOrder = line.getShortOrder();
408                    } else {
409                        checkOrder = line.getShortOrder();
410                        offsettingOrder = line.getLongOrder();
411                    }
412    
413                    checkOrderIndex = STATUSES.indexOf(checkOrder.getStatus());
414                    if (checkOrderIndex == -1) checkOrderIndex = 0;
415                    if ((STATUS_SORT_WEIGHTS[checkOrderIndex] < STATUS_SORT_WEIGHTS[orderIndex]) &&
416                        ((origOrder.getCommodityName().equals(offsettingOrder.getCommodityName())) &&
417                         (origOrder.getMonth().equals(offsettingOrder.getMonth())))) {
418                        if (origOrder.isLongOrder()) {
419                            line.setLongOrder(origOrder);
420                            if ((FILLED.equals(line.getLongOrder().getStatus()) && FILLED.equals(line.getShortOrder().getStatus())) ||
421                                (FILLED.equals(line.getLongOrder().getStatus()) && STOP_LOSS.equals(line.getShortOrder().getStatus())) ||
422                                (STOP_LOSS.equals(line.getLongOrder().getStatus()) && FILLED.equals(line.getShortOrder().getStatus()))) {
423                                line.getLongOrder().setOffsettingOrder(line.getShortOrder().getId());
424                                line.getShortOrder().setOffsettingOrder(line.getLongOrder().getId());
425                            }
426                            orderLine.setLongOrder(checkOrder);
427                            fireTableRowsUpdated(i, i);
428                            fireTableRowsUpdated(row, row);
429                        } else {
430                            line.setShortOrder(origOrder);
431                            if (FILLED.equals(line.getLongOrder().getStatus()) && FILLED.equals(line.getShortOrder().getStatus())) {
432                                line.getLongOrder().setOffsettingOrder(line.getShortOrder().getId());
433                                line.getShortOrder().setOffsettingOrder(line.getLongOrder().getId());
434                            }
435                            orderLine.setShortOrder(checkOrder);
436                            fireTableRowsUpdated(i, i);
437                            fireTableRowsUpdated(row, row);
438                        }
439                        balanceContractCount(i);
440                        balanceContractCount(row);
441                        origOrder = checkOrder;
442                    }
443                }
444            }

446            /**
447             *  Split and consolidate order lines as needed to maintain an equal number
448             *  of contract orders on both the buy and sell side of an order line.
449             *
450             *  @param  row     the row of the order line to balance
451             */
452            private void balanceContractCount(int row) {
453                ContractOrderLine   line = (ContractOrderLine)rowData.get(row);
454                ContractOrder       longOrder = line.getLongOrder();
455                ContractOrder       shortOrder = line.getShortOrder();
456                ContractOrder       newOrder = null;
457                int                 longCount = longOrder.getNumberOfContracts().intValue();
458                int                 shortCount = shortOrder.getNumberOfContracts().intValue();
459                int                 lastRowIndex = rowData.size() - 1;
460    
461                if ((longCount == shortCount) ||
462                    (longCount == 0) ||
463                    (shortCount == 0) ||
464                    (NO_STATUS.equals(longOrder.getStatus())) ||
465                    (NO_STATUS.equals(shortOrder.getStatus()))) {
466                    return;
467                }
468    
469                if (longCount > shortCount) {
470                    if ((newOrder = longOrder.split(longCount - shortCount)) == null) {
471                        return;
472                    }
473                    rowData.set(lastRowIndex, createSupportContractOrderLine(newOrder));
474                    fireTableRowsUpdated(row, row);
475                    fireTableRowsUpdated(lastRowIndex, lastRowIndex);
476                    consolidateOrders(newOrder, lastRowIndex);

478                        !(NO_STATUS.equals(((ContractOrderLine)rowData.get(lastRowIndex)).getShortOrder().getStatus()))) {

488                    consolidateOrders(newOrder, lastRowIndex);
489                    if (!(NO_STATUS.equals(((ContractOrderLine)rowData.get(lastRowIndex)).getLongOrder().getStatus())) ||
490                        !(NO_STATUS.equals(((ContractOrderLine)rowData.get(lastRowIndex)).getShortOrder().getStatus()))) {
491                        addRow();
492                    }

496            /**
497             *  Combine like orders onto a single line.  This requires that all of the
498             *  information for an order is identical including dates, contracts, prices
499             *  and statuses.
500             */
501            private void combineOrders() {
502                int startSearchRow = 0;
503    
504                for (int currentRow = (rowData.size() - 1); currentRow > 0; currentRow--) {
505                    ContractOrderLine   currentOrderLine    = (ContractOrderLine)rowData.get(currentRow);
506                    ContractOrder       currentLongOrder    = currentOrderLine.getLongOrder();
507                    ContractOrder       currentShortOrder   = currentOrderLine.getShortOrder();
508                    int currentLongValue                    = currentLongOrder.getNumberOfContracts().intValue();
509                    int currentShortValue                   = currentShortOrder.getNumberOfContracts().intValue();
510    
511                    if ((currentLongValue == 0) &&
512                        (currentShortValue == 0) &&
513                        (currentRow != (rowData.size() - 1))) {
514    
515                        removeRow(currentOrderLine);
516    
517                    } else if ((currentLongValue > 0) &&
518                               (currentShortValue == 0)) {
519    
520                        for (int searchRow = startSearchRow; searchRow < currentRow; searchRow++) {
521                            ContractOrderLine   searchOrderLine = (ContractOrderLine)rowData.get(searchRow);
522                            ContractOrder       searchLongOrder = searchOrderLine.getLongOrder();
523                            ContractOrder       searchShortOrder= searchOrderLine.getShortOrder();
524                            int searchLongValue                 = searchLongOrder.getNumberOfContracts().intValue();
525                            int searchShortValue                = searchShortOrder.getNumberOfContracts().intValue();
526    
527                            if (((searchLongValue == 0) ||
528                                 (searchShortValue == 0)) &&
529                                (startSearchRow == 0)) {
530                                startSearchRow = searchRow;
531                            }
532    
533                            if ((searchShortValue == 0) &&
534                                (currentLongOrder.equals(searchLongOrder))) {
535                                searchLongOrder.combine(currentLongOrder);
536                                fireTableRowsUpdated(searchRow, searchRow);
537                                removeRow(currentOrderLine);
538                                break;
539                            }
540                        }
541    
542                    } else if ((currentLongValue == 0) &&
543                               (currentShortValue > 0)) {
544    
545                        for (int searchRow = startSearchRow; searchRow < currentRow; searchRow++) {
546                            ContractOrderLine   searchOrderLine = (ContractOrderLine)rowData.get(searchRow);
547                            ContractOrder       searchLongOrder = searchOrderLine.getLongOrder();
548                            ContractOrder       searchShortOrder= searchOrderLine.getShortOrder();
549                            int searchLongValue                 = searchLongOrder.getNumberOfContracts().intValue();
550                            int searchShortValue                = searchShortOrder.getNumberOfContracts().intValue();
551    
552                            if (((searchLongValue == 0) ||
553                                 (searchShortValue == 0)) &&
554                                (startSearchRow == 0)) {
555                                startSearchRow = searchRow;
556                            }
557    
558                            if ((searchLongValue == 0) &&
559                                (currentShortOrder.equals(searchShortOrder))) {
560                                searchShortOrder.combine(currentShortOrder);
561                                fireTableRowsUpdated(searchRow, searchRow);
562                                removeRow(currentOrderLine);
563                                break;
564                            }
565                        }
566                    } else if ((currentLongValue > 0) &&
567                               (currentShortValue > 0)) {
568                        for (int searchRow = 0; searchRow < currentRow; searchRow++) {
569                            ContractOrderLine   searchOrderLine = (ContractOrderLine)rowData.get(searchRow);
570                            ContractOrder       searchLongOrder = searchOrderLine.getLongOrder();
571                            ContractOrder       searchShortOrder= searchOrderLine.getShortOrder();
572    
573                            if (currentLongOrder.equals(searchLongOrder) &&
574                                currentShortOrder.equals(searchShortOrder)) {
575                                searchLongOrder.combine(currentLongOrder);
576                                searchShortOrder.combine(currentShortOrder);
577                                fireTableRowsUpdated(searchRow, searchRow);
578                                removeRow(currentOrderLine);
579                                break;
580                            }
581                        }
582                    }
583                }
584            }
