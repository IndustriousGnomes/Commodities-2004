003    package commodities.util;
004    
005    import java.util.*;
006    import com.util.GregorianWeekdayCalendar;
007    
008    /**
009     *  The CommodityCalendar class extends the gregorian weekday calendar
010     *  to override normal calendar functions for commodity processing.
011     *
012     *  @author J.R. Titko
013     *  @version 1.0
014     *  @update 2004.11.11
015     */
016    
017    public class CommodityCalendar extends GregorianWeekdayCalendar {
018    /* *************************************** */
019    /* *** CLASS VARIABLES                 *** */
020    /* *************************************** */
021        private static int alternateYear = -1;
022        private static int alternateMonth = -1;
023        private static int alternateDate = -1;
024    
025    /* *************************************** */
026    /* *** CONSTRUCTORS                    *** */
027    /* *************************************** */
028        /**
029         *  Constructs a default CommodityCalendar using the current time in the
030         *  default time zone with the default locale.
031         */
032        public CommodityCalendar() {
033            super();
034    //        this(2003, 9, 29);
035        }
036    
037        /**
038         *  Constructs a CommodityCalendar with the given date set in the default
039         *  time zone with the default locale.
040         */
041        public CommodityCalendar(Date date) {
042            super();
043            if (date != null) {
044                setTime(date);
045            }
046        }
047    
048        /**
049         *  Constructs a CommodityCalendar with the given date set in the default
050         *  time zone with the default locale.
051         */
052        public CommodityCalendar(int year, int month, int date) {
053            super(year, month, date);
054        }
055    
056        /**
057         *  Constructs a CommodityCalendar with the given date and time set for
058         *  the default time zone with the default locale.
059         */
060        public CommodityCalendar(int year, int month, int date, int hour, int minute) {
061            super(year, month, date, hour, minute);
062        }
063    
064        /**
065         *  Constructs a CommodityCalendar with the given date and time set for
066         *  the default time zone with the default locale.
067         */
068        public CommodityCalendar(int year, int month, int date, int hour, int minute, int second) {
069            super(year, month, date, hour, minute, second);
070        }
071    
072        /**
073         *  Constructs a CommodityCalendar based on the current time in the
074         *  default time zone with the given locale.
075         */
076        public CommodityCalendar(Locale aLocale) {
077            super(aLocale);
078        }
079    
080        /**
081         *  Constructs a CommodityCalendar based on the current time in the
082         *  given time zone with the default locale.
083         */
084        public CommodityCalendar(TimeZone zone) {
085            super(zone);
086        }
087    
088        /**
089         *  Constructs a CommodityCalendar based on the current time in the
090         *  given time zone with the given locale.
091         */
092        public CommodityCalendar(TimeZone zone, Locale aLocale) {
093            super(zone, aLocale);
094        }
095    
096    
097    /* *************************************** */
098    /* *** Calendar METHODS                *** */
099    /* *************************************** */
100        /**
101         *  Gets a calendar using the default time zone and locale or the simulated
102         *  date. The Calendar returned is based on the current time in the default
103         *  time zone with the default locale.
104         *
105         *  If the simulated date has been set, but not cleared, the returned calendar
106         *  will be set to the simulator date.
107         *
108         *  @return     a calendar
109         */
110        public static Calendar getInstance() {
111            if (alternateYear == -1) {
112                return new CommodityCalendar();
113            } else {
114                return new CommodityCalendar(alternateYear, alternateMonth, alternateDate);
115            }
116        }
117    
118    /* *************************************** */
119    /* *** INSTANCE METHODS                *** */
120    /* *************************************** */
121        /**
122         *  Sets the calendar date for the Commodities application to the current
123         *  date for this calendar.
124         */
125        public void setSimulatorDate() {
126            alternateYear    = get(Calendar.YEAR);
127            alternateMonth   = get(Calendar.MONTH);
128            alternateDate    = get(Calendar.DATE);
129        }
130    
131        /**
132         *  Sets the calendar date for the Commodities application to today's date.
133         */
134        public void clearSimulatorDate() {
135            alternateYear    = -1;
136            alternateMonth   = -1;
137            alternateDate    = -1;
138        }
139    }
