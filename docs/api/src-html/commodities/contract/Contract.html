101        /**
102         *  LoadThread loads the contract prices
103         *  as a thread so the main processing can continue.
104         *
105         *  @author J.R. Titko
106         */
107        public class LoadThread implements Runnable {
108        /* *************************************** */
109        /* *** Thread METHODS                  *** */
110        /* *************************************** */
111            public void run() {
112                try {
113                    priceListStatus.pos2();     // mark price list as loading
114                    Iterator it = dataManager.getPrices(id);
115                    while (it.hasNext()) {
116                        Prices prices = (Prices)it.next();
117    
118                        if ((currentPrices == null) || (prices.getDate().compareTo(currentPrices.getDate()) > 0)) {
119                            currentPrices = prices;
120                        }
121    
122                        if (prices.getHigh() > maxPrice) {
123                            maxPrice = prices.getHigh();
124                        }
125                        if (prices.getLow() < minPrice) {
126                            minPrice = prices.getLow();
127                        }
128                        priceList.put(prices.getDate(), prices);
129                    }
130                } catch (IOException e) {
131                    e.printStackTrace();
132                }
133    //System.out.println("firePricesUpdated #1");
134                priceListStatus.pos3();     // mark price list as loaded
135                firePricesUpdated(getCurrentDate());
136            }
137        }

367         *  @param  date    The date to retrieve the prices for.
368         *  @return     the Prices for the given date.
369         */
370        public synchronized Prices getPrices(java.util.Date date) {
371    if (!priceListStatus.isPos3()) {
372        System.out.println("Contract.getPrices() - delayed");
373    }
374            if (currentPrices == null) {
375                getPrices();
376            }
377            while (!priceListStatus.isPos3()) {
378                synchronized(this) {
379                    try {
380                        wait(10);
381                    } catch (InterruptedException e) {
382                    }
383                }
384            }
385            if ((priceList == null) || (priceList.size() == 0)) {
386                return null;
387            } else {
388                return (Prices)priceList.get(date);
389            }
390        }

392        /**
393         *  Get the closest Prices for the given date of this contract.  If date
394         *  has a price, then that date's prices will be returned.  If date
395         *  does not have prices, then the previous price closes to that date
396         *  is returned.
397         *
398         *  @param  date    the date to retrieve the prices for
399         *  @return     the nearest Prices to the date
400         */
401        public synchronized Prices getNearestPrices(java.util.Date date) {
402            Prices prices = getPrices(date);
403    
404            if (prices == null && date != null) {
405                int countDown = 7;
406                CommodityCalendar calendar = new CommodityCalendar(date);
407                do {
408                    calendar.addWeekDays(-1);
409                    prices = getPrices(calendar.getTime());
410                } while ((prices == null) || (countDown-- > 0));
411            }
412            return prices;
413        }

463        /**
464         *  Get all of the Price date keys in a ListIterator format
465         *  for forwards and backwards traversal of the list.
466         *
467         *  A descending (Contract.DESCENDING) list will begin at the most current date and
468         *  should be traversed backwards (previous).
469         *  An ascending list (Contract.ASCENDING) will begin at the earliest
470         *  available date and should be traversed forwards (next).
471         *
472         *  This ListIterator cannot have objects added or removed from it.
473         *
474         *  @param  order   The order (ascending or decending) to return the list in.
475         *  @return the ListIterator of keys.
476         */
477        public synchronized ListIterator getPriceDates(boolean order) {
478            if (order) {
479                return priceList.listIterator(priceList.size());
480            } else {
481                return priceList.listIterator(0);
482            }
483        }

690        /**
691         *  Fire all price information to all PriceListeners.
692         *
693         *  @param  date  the date of prices to display
694         */
695        public void firePricesUpdated(Prices prices) {
696            PriceEvent dateEvent    = new PriceEvent(this, prices.getDate(), prices.getDate());
697            PriceEvent openEvent    = new PriceEvent(this, prices.getDate(), prices.getOpenObject());
698            PriceEvent highEvent    = new PriceEvent(this, prices.getDate(), prices.getHighObject());
699            PriceEvent lowEvent     = new PriceEvent(this, prices.getDate(), prices.getLowObject());
700            PriceEvent closeEvent   = new PriceEvent(this, prices.getDate(), prices.getCloseObject());
701            PriceEvent interestEvent = new PriceEvent(this, prices.getDate(), prices.getInterestObject());
702            PriceEvent volumeEvent  = new PriceEvent(this, prices.getDate(), prices.getVolumeObject());
703    
704            Iterator it = priceListeners.iterator();
705            while (it.hasNext()) {
706                PriceListener listener = (PriceListener)it.next();
707                listener.changeDate(dateEvent);
708                listener.changeOpenPrice(openEvent);
709                listener.changeHighPrice(highEvent);
710                listener.changeLowPrice(lowEvent);
711                listener.changeClosePrice(closeEvent);
712                listener.changeOpenInterest(interestEvent);
713                listener.changeVolume(volumeEvent);
714            }
715            lastFiredDate = null;
716        }

761         *  @param  formattedMonth  month in 'MMM YY'

763         */
764        public static String convertFormattedMonthToKey(String formattedMonth) {
765            StringBuffer formatted = new StringBuffer(formattedMonth);
766            StringBuffer key = new StringBuffer();
767    
768            key.append("20");
769            key.append(formatted.substring(formatted.length() - 2));
770            key.append(Month.byAbbrev(formatted.substring(0, 3)).getSymbol());
771            return key.toString();
772        }

776         *  a listener is not adequate.

815        /**
816         *  Fire a contract selection change.
817         *
818         *  @param  prices  the prices to display
819         */
820        public static void fireSelectionUpdated() {
821            ContractSelectionEvent event = new ContractSelectionEvent(selectedContract);
822    
823            Iterator it = selectionListeners.iterator();
824            while (it.hasNext()) {
825                ContractSelectionListener listener = (ContractSelectionListener)it.next();
826                listener.selectContract(event);
827            }
828        }
