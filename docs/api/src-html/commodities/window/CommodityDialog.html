129        /**
130         *  Set up the GUI interface of the commodity dialog box.
131         */
132        private void setupGUI() {
133            setResizable(false);
134            Dimension screenSize = CommodityAnalyzerJFrame.instance().getBounds().getSize();
135            setLocation(((screenSize.width - DIALOG_WIDTH) / 2), ((screenSize.height - DIALOG_HEIGHT) / 3));
136    
137            acceptBtn.addActionListener(new ActionListener() {
138                public void actionPerformed(ActionEvent event) {
139                    try {
140                        if (!"".equals(symbolTF.getText()) &&
141                            !"".equals(exchangeTF.getText()) &&
142                            !"".equals(nameTF.getText()) &&
143                            !"".equals(unitTypeTF.getText()) &&
144                            !"".equals(unitSizeTF.getText()) &&
145                            !"".equals(unitPriceTF.getText()) &&
146                            !"".equals(tickSizeTF.getText()) &&
147                            !"".equals(tickPriceTF.getText()) &&
148                            !"".equals(tickDailyLimitTF.getText()) &&
149                            !"".equals(marginTF.getText()) &&
150                            !"".equals(decimalsTF.getText()) &&
151                            !"".equals(monthsTF.getText())) {
152    
153                            int monthMask = formatMonthMask(monthsTF.getText());
154    
155                            if (commodity == null) {
156                                commodity = new Commodity(symbolTF.getText(),
157                                                          exchangeTF.getText(),
158                                                          nameTF.getText(),
159                                                          unitTypeTF.getText(),
160                                                          Integer.parseInt(unitSizeTF.getText()),
161                                                          unitPriceTF.getText(),
162                                                          Double.parseDouble(tickSizeTF.getText()),
163                                                          Double.parseDouble(tickPriceTF.getText()),
164                                                          Integer.parseInt(tickDailyLimitTF.getText()),
165                                                          Integer.parseInt(marginTF.getText()),
166                                                          Integer.parseInt(decimalsTF.getText()),
167                                                          openOutcrySymbolTF.getText(),
168                                                          acmSymbolTF.getText(),
169                                                          monthMask,
170                                                          databaseSymbolTF.getText());
171                                try {
172                                    dataManager.addCommodity(commodity);
173                                } catch (IOException e) {
174                                    e.printStackTrace();
175                                }
176                            } else {
177                                commodity.setName(nameTF.getText());
178                                commodity.setUnitType(unitTypeTF.getText());
179                                commodity.setUnitSize(Integer.parseInt(unitSizeTF.getText()));
180                                commodity.setUnitPrice(unitPriceTF.getText());
181                                commodity.setTickSize(Double.parseDouble(tickSizeTF.getText()));
182                                commodity.setTickPrice(Double.parseDouble(tickPriceTF.getText()));
183                                commodity.setTickDailyLimit(Integer.parseInt(tickDailyLimitTF.getText()));
184                                commodity.setMargin(Integer.parseInt(marginTF.getText()));
185                                commodity.setDisplayDecimals(Integer.parseInt(decimalsTF.getText()));
186                                commodity.setOpenOutcrySymbol(openOutcrySymbolTF.getText());
187                                commodity.setAcmSymbol(acmSymbolTF.getText());
188                                commodity.setTradeMonthMask(monthMask);
189                                commodity.setDatabaseSymbol(databaseSymbolTF.getText());
190    
191                                try {
192                                    dataManager.updateCommodity(commodity);
193                                } catch (IOException e) {
194                                    e.printStackTrace();
195                                }
196                            }
197    
198                            setVisible(false);
199                            dispose();
200                        } else {
201                            JOptionPane.showMessageDialog(CommodityAnalyzerJFrame.instance(), "Not all required fields are entered.");
202                        }
203                    } catch (NumberFormatException ex) {
204                        JOptionPane.showMessageDialog(CommodityAnalyzerJFrame.instance(), "Fields are not propertly formatted.");
205                    }
206                }
207            });
208    
209            cancelBtn.addActionListener(new ActionListener() {
210                public void actionPerformed(ActionEvent event) {
211                    setVisible(false);
212                    dispose();
213                }
214            });
215    
216            getRootPane().setDefaultButton(acceptBtn);
217    
218            JPanel inputPanel = new JPanel();
219            inputPanel.setLayout(new FlowLayout());
220            inputPanel.setPreferredSize(new Dimension(DIALOG_WIDTH, DIALOG_HEIGHT));
221    
222            if (commodity != null) {
223                symbolTF.setText(commodity.getSymbol());
224                symbolTF.setEditable(false);
225                exchangeTF.setText(commodity.getExchange());
226                exchangeTF.setEditable(false);
227    
228                nameTF.setText(commodity.getName());
229                unitTypeTF.setText(commodity.getUnitType());
230                unitSizeTF.setText(""+commodity.getUnitSize());
231                unitPriceTF.setText(commodity.getUnitPrice());
232                tickSizeTF.setText(""+commodity.getTickSize());
233                tickPriceTF.setText(""+commodity.getTickPrice());
234                tickDailyLimitTF.setText(""+commodity.getTickDailyLimit());
235                marginTF.setText(""+commodity.getMargin());
236                decimalsTF.setText(""+commodity.getDisplayDecimals());
237                monthsTF.setText(formatMonths(commodity.getTradeMonthMask()));
238                openOutcrySymbolTF.setText(commodity.getOpenOutcrySymbol());
239                acmSymbolTF.setText(commodity.getAcmSymbol());
240                databaseSymbolTF.setText(commodity.getDatabaseSymbol());
241            }
242    
243            inputPanel.add(new FieldPanel(symbolTF, "Symbol (ex: S)"));
244            inputPanel.add(new FieldPanel(exchangeTF, "Exchange (ex: CBOT)"));
245            inputPanel.add(new FieldPanel(nameTF, "Name (ex: Soybeans)"));
246            inputPanel.add(new FieldPanel(unitTypeTF, "Unit Type  (ex: bu)"));
247            inputPanel.add(new FieldPanel(unitSizeTF, "Unit Size (ex: 5000)"));
248            inputPanel.add(new FieldPanel(unitPriceTF, "Unit Price  (cents)"));
249            inputPanel.add(new FieldPanel(tickSizeTF, "Tick Size (ex: 0.25)"));
250            inputPanel.add(new FieldPanel(tickPriceTF, "Tick Price (ex: 12.50)"));
251            inputPanel.add(new FieldPanel(tickDailyLimitTF, "Tick Daily Limit (ex: 200)"));
252            inputPanel.add(new FieldPanel(marginTF, "Margin (ex: 1500)"));
253            inputPanel.add(new FieldPanel(decimalsTF, "Decimals to Display (ex: 2)"));
254            inputPanel.add(new FieldPanel(monthsTF, "Months Traded (ex: JAN, FEB)"));
255            inputPanel.add(new FieldPanel(openOutcrySymbolTF, "Open Outcry Symbol (opt)"));
256            inputPanel.add(new FieldPanel(acmSymbolTF, "ACM Symbol (opt)"));
257            inputPanel.add(new FieldPanel(databaseSymbolTF, "Database Symbol (opt)"));
258    
259    
260            JPanel buttonPanel = new JPanel();
261            buttonPanel.setLayout(new BoxLayout(buttonPanel, BoxLayout.X_AXIS));
262            buttonPanel.setBorder(BorderFactory.createEmptyBorder(0, 10, 10, 10));
263            buttonPanel.add(Box.createHorizontalGlue());
264            buttonPanel.add(acceptBtn);
265            buttonPanel.add(Box.createHorizontalStrut(10));
266            buttonPanel.add(cancelBtn);
267            buttonPanel.add(Box.createHorizontalGlue());
268    
269            Container contentPane = getContentPane();
270            contentPane.add(inputPanel, BorderLayout.CENTER);
271            contentPane.add(buttonPanel, BorderLayout.SOUTH);
272    
273            pack();
274            setVisible(true);
275        }

277        /**
278         *  Format the names of the months based on the month mask for the commodity.
279         *  The names are formatted using a comma dilimited list of 3 char month names.
280         *
281         *  @param  monthMask   The binary month mask for the commodity trading months
282         *  @return             The list of trading months in string format
283         */
284        private String formatMonths(int monthMask) {
285            String months = "";
286            boolean first = true;
287            if (!"".equals(months += formatMonth(monthMask, Commodity.JAN, "JAN", first))) { first = false; };
288            if (!"".equals(months += formatMonth(monthMask, Commodity.FEB, "FEB", first))) { first = false; };
289            if (!"".equals(months += formatMonth(monthMask, Commodity.MAR, "MAR", first))) { first = false; };
290            if (!"".equals(months += formatMonth(monthMask, Commodity.APR, "APR", first))) { first = false; };
291            if (!"".equals(months += formatMonth(monthMask, Commodity.MAY, "MAY", first))) { first = false; };
292            if (!"".equals(months += formatMonth(monthMask, Commodity.JUN, "JUN", first))) { first = false; };
293            if (!"".equals(months += formatMonth(monthMask, Commodity.JUL, "JUL", first))) { first = false; };
294            if (!"".equals(months += formatMonth(monthMask, Commodity.AUG, "AUG", first))) { first = false; };
295            if (!"".equals(months += formatMonth(monthMask, Commodity.SEP, "SEP", first))) { first = false; };
296            if (!"".equals(months += formatMonth(monthMask, Commodity.OCT, "OCT", first))) { first = false; };
297            if (!"".equals(months += formatMonth(monthMask, Commodity.NOV, "NOV", first))) { first = false; };
298            if (!"".equals(months += formatMonth(monthMask, Commodity.DEC, "DEC", first))) { first = false; };
299            return months;
300        }
301    
302        /**
303         *  Perform the formatting logic for a single month.
304         *
305         *  @param  monthMask   The binary month mask for the commodity trading months
306         *  @param  mask        The binary month mask for a single month
307         *  @param  month       The month string to be used if found in the month mask
308         *  @param  first       Is this the first month found for the month string
309         *  @return             The list of trading months in string format
310         */
311        private String formatMonth(int monthMask, int mask, String month, boolean first) {
312            String label = "";
313            if ((monthMask & mask) > 0) {
314                label = first?month:", " + month;
315            }
316            return label;
317        }
318    
319        /**
320         *  Format the month mask based on the 3 character month names.
321         *
322         *  @param  monthMask   The list of trading months in string format
323         *  @return             The binary month mask for the commodity trading months
324         */
325        private int formatMonthMask(String monthText) {
326            int monthMask = 0;
327            monthText = monthText.toUpperCase();
328    
329            if (monthText.indexOf("JAN") > -1) { monthMask += Commodity.JAN; }
330            if (monthText.indexOf("FEB") > -1) { monthMask += Commodity.FEB; }
331            if (monthText.indexOf("MAR") > -1) { monthMask += Commodity.MAR; }
332            if (monthText.indexOf("APR") > -1) { monthMask += Commodity.APR; }
333            if (monthText.indexOf("MAY") > -1) { monthMask += Commodity.MAY; }
334            if (monthText.indexOf("JUN") > -1) { monthMask += Commodity.JUN; }
335            if (monthText.indexOf("JUL") > -1) { monthMask += Commodity.JUL; }
336            if (monthText.indexOf("AUG") > -1) { monthMask += Commodity.AUG; }
337            if (monthText.indexOf("SEP") > -1) { monthMask += Commodity.SEP; }
338            if (monthText.indexOf("OCT") > -1) { monthMask += Commodity.OCT; }
339            if (monthText.indexOf("NOV") > -1) { monthMask += Commodity.NOV; }
340            if (monthText.indexOf("DEC") > -1) { monthMask += Commodity.DEC; }
341            return monthMask;
342        }
