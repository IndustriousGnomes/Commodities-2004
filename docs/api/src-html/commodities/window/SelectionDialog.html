003    package commodities.window;
004    
005    import java.awt.*;
006    import java.awt.event.*;
007    import javax.swing.*;
008    
009    /**
010     *  The SelectionDialog class creates a dialog box containing
011     *  selections in the form of a list.
012     *
013     *  @author J.R. Titko
014     *  @version 1.0
015     *  @update 2004.11.11
016     */
017    
018    public class SelectionDialog extends JDialog {
019    /* *************************************** */
020    /* *** INSTANCE VARIABLES              *** */
021    /* *************************************** */
022    //    private Frame frame;
023        /** List of items to be selected from in the dialog box. */
024        private JList list;
025        /** The selection mode of the list: single or multiple selection.  */
026        private int selectionMode;
027        /** The list of choices passed in as an array. */
028        private Object choices[];
029        /** The list of selected entities from the list as an array. */
030        private Object selected[];
031        /** The current instance */
032    //    private SelectionDialog myDialog;
033    
034        private JButton acceptBtn;
035        private JButton cancelBtn = new JButton("Cancel");
036    
037    /* *************************************** */
038    /* *** CONSTRUCTORS                    *** */
039    /* *************************************** */
040        public SelectionDialog(Frame frame, String title, Object choices[], String acceptButtonName, int selectionMode) {
041            super(frame, title, true);
042    //        this.frame = frame;
043            this.choices = choices;
044            this.selectionMode = selectionMode;
045            acceptBtn = new JButton(acceptButtonName);
046    
047    //        myDialog = this;
048            setupGUI();
049        }
050    
051        public SelectionDialog(Frame frame, String title, Object choices[], int selectionMode) {
052            this(frame, title, choices, "Ok", selectionMode);
053        }
054    
055        public SelectionDialog(Frame frame, String title, Object choices[], String acceptButtonName) {
056            this(frame, title, choices, "Ok", ListSelectionModel.MULTIPLE_INTERVAL_SELECTION);
057        }
058    
059        public SelectionDialog(Frame frame, String title, Object choices[]) {
060            this(frame, title, choices, "Ok", ListSelectionModel.MULTIPLE_INTERVAL_SELECTION);
061        }
062    
063    /* *************************************** */
064    /* *** INSTANCE METHODS                *** */
065    /* *************************************** */
066        private void setupGUI() {
067            acceptBtn.addActionListener(new ActionListener() {
068                public void actionPerformed(ActionEvent e) {
069                    selected = list.getSelectedValues();
070    //                myDialog.setVisible(false);
071                    setVisible(false);
072    //                myDialog.dispose();
073                    dispose();
074                }
075            });
076    
077            cancelBtn.addActionListener(new ActionListener() {
078                public void actionPerformed(ActionEvent e) {
079    //                myDialog.setVisible(false);
080    //                myDialog.dispose();
081                    setVisible(false);
082                    dispose();
083                }
084            });
085    
086            getRootPane().setDefaultButton(acceptBtn);
087    
088            list = new JList(choices);
089            list.setSelectionMode(selectionMode);
090            list.addMouseListener(new MouseAdapter() {
091                public void mouseClicked(MouseEvent e) {
092                    if (e.getClickCount() == 2) {
093                        acceptBtn.doClick();
094                    }
095                }
096            });
097    
098            JScrollPane jsp = new JScrollPane(list);
099            jsp.setPreferredSize(new Dimension(250, 80));
100            jsp.setMinimumSize(new Dimension(250, 80));
101    
102            JPanel listPanel = new JPanel();
103            listPanel.add(jsp);
104            listPanel.setBorder(BorderFactory.createEmptyBorder(10,5,5,5));
105    
106    
107            JPanel buttonPanel = new JPanel();
108            buttonPanel.setLayout(new BoxLayout(buttonPanel, BoxLayout.X_AXIS));
109            buttonPanel.setBorder(BorderFactory.createEmptyBorder(0, 10, 10, 10));
110            buttonPanel.add(Box.createHorizontalGlue());
111            buttonPanel.add(acceptBtn);
112            buttonPanel.add(Box.createHorizontalStrut(10));
113            buttonPanel.add(cancelBtn);
114            buttonPanel.add(Box.createHorizontalGlue());
115    
116            Container contentPane = getContentPane();
117            contentPane.add(listPanel, BorderLayout.CENTER);
118            contentPane.add(buttonPanel, BorderLayout.SOUTH);
119    
120    //        setLocationRelativeTo(frame);
121            pack();
122            setVisible(true);
123    
124        }
125    
126        public Object[] getSelected() {
127            return selected;
128        }
129    
130    
131    }
