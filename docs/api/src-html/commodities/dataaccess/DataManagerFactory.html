003    package commodities.dataaccess;
004    
005    import commodities.dataaccess.database.*;
006    import commodities.util.*;
007    
008    /**
009     *  The DataManagerFactory decides which implementation
010     *  of the data manager is being used.
011     *
012     *  The name of the datamanager is located in the
013     *  properties file identified as dataaccess.
014     *
015     *  @author J.R. Titko
016     *  @version 1.0
017     *  @update 2004.11.11
018     */
019    
020    public class DataManagerFactory {
021    /* *************************************** */
022    /* *** CLASS VARIABLES                 *** */
023    /* *************************************** */
024        /** Properties file identifier for the properties manager. */
025        private static final String DATA_ACCESS_PROPERTIES_FILE = "dataaccess";
026        /** Property for identifying the properties manager. */
027        private static final String DATA_MANAGER = "datamanager";
028        /** SQL Server data manager */
029        public static final String SQLSERVER = "SQLServer";
030        /** Access data manager */
031        public static final String ACCESS = "Access";
032    
033        /** The datamanager to use. */
034       private static String dataManager;
035    
036    /* *************************************** */
037    /* *** STATIC METHODS                  *** */
038    /* *************************************** */
039        /**
040         *  Retrieve the data manager.
041         *
042         *  @return     the active data manager
043         */
044        public static DataManagerInterface instance() {
045            if (dataManager == null) {
046                synchronized (DataManagerFactory.class) {
047                    if (dataManager == null) {
048                       dataManager = PropertyManager.instance().getProperties(DATA_ACCESS_PROPERTIES_FILE).getProperty(DATA_MANAGER);
049                    }
050                }
051            }
052            if (SQLSERVER.equals(dataManager)) {
053                return SQLServerDataManager.instance();
054            } else if (ACCESS.equals(dataManager)) {
055                return AccessDataManager.instance();
056            }
057    
058            return null;
059        }
060    }
