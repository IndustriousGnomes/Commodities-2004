003    package commodities.event;
004    
005    import java.awt.*;
006    import java.util.*;
007    
008    /**
009     *  The GraphRedrawListener is an EventListener that controls the drawing
010     *  of graphs.  It is responsible for supplying the title of the graph
011     *  and the contents of the graph.  Although it is used to supply the
012     *  information for calculating the graph's scale, the actual scale is
013     *  in the GraphRedrawEvent as multiple graphs can occupy the same drawing
014     *  panel.
015     *
016     *  @see    commodities.graph.Graph
017     *  @author J.R. Titko
018     *  @version 1.0
019     *  @update 2004.11.11
020     */
021    
022    public interface GraphRedrawListener extends EventListener {
023        /**
024         *  Gets the number of pixels that should be added as white
025         *  space to the top and bottom of the graph when calculating
026         *  the scale of the graph.  This value is inside the graph's
027         *  frame and therefore reduces the amount of drawing space
028         *  by twice its amount.
029         *
030         *  @return         The number of pixels to use at the
031         *                  top and bottom of the graph.
032         */
033        public double getYBuffer();
034        /**
035         *  Gets the maximum value represented in the graph.
036         *
037         *  @return         The data's maximum value.
038         */
039        public double getMaxValue();
040        /**
041         *  Gets the minimum value represented in the graph.
042         *
043         *  @return         The data's minimum value.
044         */
045        public double getMinValue();
046        /**
047         *  Gets the size of a tick for the commodity being drawn.
048         *  This keeps the scale of the graph in multiples of this
049         *  tick size.
050         *
051         *  @return         The data's tick size.
052         */
053        public double getTickSize();
054        /**
055         *  Defines how the title for the graph is to be drawn.
056         *  The yOffset is used if multiple titles appear in
057         *  the same graph to offset them so they dont write over
058         *  each other.
059         *
060         *  @param  g       The graphics area to draw in.
061         *  @param  yOffset The y-axis offset the title should have.
062         *  @return         True if a title was drawn.
063         */
064        public boolean redrawTitle(Graphics g, int yOffset);
065        /**
066         *  Draws the graph using the GraphRedrawEvent.  This is limited
067         *  to the statistical part of the graph and not the graph frame
068         *  or scale indicators.
069         *
070         *  @param  e       The GraphRedrawEvent
071         */
072        public void   redrawComponent(GraphRedrawEvent e);
073    }
