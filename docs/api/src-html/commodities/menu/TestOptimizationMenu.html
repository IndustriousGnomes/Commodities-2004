003    package commodities.menu;
004    
005    import java.awt.event.*;
006    import java.util.*;
007    import javax.swing.*;
008    
009    import commodities.contract.*;
010    import commodities.price.*;
011    import commodities.tests.*;
012    import commodities.window.*;
013    
014    /**
015     *  The TestOptimizationMenu class determines what tests to run the optimizer for.
016     *
017     *  @author J.R. Titko
018     *  @version 1.0
019     *  @update 2004.11.15
020     */
021    
022    public class TestOptimizationMenu extends JMenu implements ActionListener {
023    /* *************************************** */
024    /* *** CLASS VARIABLES                 *** */
025    /* *************************************** */
026        /** Menu entry name to optimize the tests for the selected contract. */
027        public static final String OPTIMIZE_SELECTED_CONTRACT   = "Selected Contract";
028        /** Menu entry name to optimize the tests for all contracts. */
029        public static final String OPTIMIZE_SELECTED_COMMODITY  = "Selected Commodity";
030        /** Menu entry name to optimize the tests for all contracts. */
031        public static final String OPTIMIZE_ALL_COMMODITIES     = "All Commodities";
032        /** Reference to the TestOptimizationMenu instance */
033        private static TestOptimizationMenu menu = null;
034    
035    /* *************************************** */
036    /* *** INSTANCE VARIABLES              *** */
037    /* *************************************** */
038    
039    /* *************************************** */
040    /* *** CONSTRUCTORS                    *** */
041    /* *************************************** */
042        /**
043         *  Creates the Tests menu.
044         */
045        public TestOptimizationMenu() {
046           super("Optimize Tests");
047            setupMenu();
048            menu = this;
049        }
050    
051    /* *************************************** */
052    /* *** METHODS                         *** */
053    /* *************************************** */
054        /**
055         *  Set up the Test menu.
056         */
057        private void setupMenu() {
058            JMenuItem menuItem = null;
059    
060            menuItem = new JMenuItem(OPTIMIZE_SELECTED_CONTRACT);
061            menuItem.addActionListener(this);
062            add(menuItem);
063    
064            menuItem = new JMenuItem(OPTIMIZE_SELECTED_COMMODITY);
065            menuItem.addActionListener(this);
066            add(menuItem);
067    
068            menuItem = new JMenuItem(OPTIMIZE_ALL_COMMODITIES);
069            menuItem.addActionListener(this);
070            add(menuItem);
071        }
072    
073    /* *************************************** */
074    /* *** ActionListener INTERFACE        *** */
075    /* *************************************** */
076        /**
077         *  Handle a selection in the menu.
078         *
079         *  @param  e   the ItemEvent
080         */
081        public void actionPerformed(ActionEvent e) {
082            JMenuItem source = (JMenuItem)e.getSource();
083            String menuEntry = source.getText();
084    
085            if (OPTIMIZE_SELECTED_CONTRACT.equals(menuEntry)) {
086                try {
087                    new TestOptimizer(Contract.getSelectedContract());
088                } catch (ContractNotSelectedException ex) {
089                    JOptionPane.showMessageDialog(CommodityAnalyzerJFrame.instance(), "No contract is currently selected.  Please select a contract first.");
090                }
091            } else if (OPTIMIZE_SELECTED_COMMODITY.equals(menuEntry)) {
092                try {
093                    new TestOptimizer(Contract.getSelectedContract().getCommodity());
094                } catch (ContractNotSelectedException ex) {
095                    JOptionPane.showMessageDialog(CommodityAnalyzerJFrame.instance(), "No contract in a commodity is currently selected.  Please select a contract first.");
096                }
097            } else if (OPTIMIZE_ALL_COMMODITIES.equals(menuEntry)) {
098                new TestOptimizer();
099            }
100        }
101    }
