003    package commodities.menu;
004    
005    import java.awt.event.*;
006    import javax.swing.*;
007    
008    /**
009     *  The FileMenu class controls the actions listed in the File menu.
010     *
011     *  @author J.R. Titko
012     *  @version 1.0
013     *  @update 2004.11.11
014     */
015    public class FileMenu extends JMenu implements ActionListener {
016    /* *************************************** */
017    /* *** CLASS VARIABLES                 *** */
018    /* *************************************** */
019        /** Text for the Quit menu option. */
020        private static final String QUIT = "Quit";
021    
022    /* *************************************** */
023    /* *** CONSTRUCTORS                    *** */
024    /* *************************************** */
025        /**
026         *  Create the File menu.
027         */
028        public FileMenu() {
029            super("File");
030            setupMenu();
031        }
032    
033    /* *************************************** */
034    /* *** METHODS                         *** */
035    /* *************************************** */
036        /**
037         *  Setup the menu with its options.
038         */
039        private void setupMenu() {
040            JMenuItem menuItem = null;
041    
042            menuItem = new JMenuItem(QUIT);
043            menuItem.addActionListener(this);
044            add(menuItem);
045    
046        }
047    
048    
049    /* *************************************** */
050    /* *** ActionListener INTERFACE        *** */
051    /* *************************************** */
052        /**
053         *  Handle the selection of menu options.
054         */
055        public void actionPerformed(ActionEvent e) {
056            JMenuItem source = (JMenuItem)(e.getSource());
057            if (QUIT.equals(source.getText())) {
058                System.exit(0);
059            } else {
060                String s = "Action event detected."
061                           + "\n"
062                           + "    Event source: " + source.getText()
063                           + " (an instance of " + source.getClass().getName() + ")";
064                System.out.println(s + "\n");
065            }
066        }
067    }
