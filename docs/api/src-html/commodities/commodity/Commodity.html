490        /**
491         *  Adds a new contract to the commodity.  This contract will be stored.
492         *
493         *  @param  contractMonth    the month of the contract in 'YYYYM'
494         *  @return     the contract added or the original contract
495         */
496        public Contract addContract(String contractMonth) {
497            Contract contract = null;
498            if ((contract = getContract(contractMonth)) == null) {
499                try {
500                    dataManager.addContract(symbol, contractMonth);
501                    contract = dataManager.getContractByMonth(symbol, contractMonth);
502                    addContract(contract);
503                } catch (IOException e) {
504                    System.out.println("Failed to add contract " + symbol + " " + contractMonth);
505                    return null;
506                }
507            }
508            return contract;
509        }

518            contracts.put(contract.getMonth(), contract);

521        /**
522         *  Load the contracts that go with this commodity into memory.
523         */
524        private void loadContracts() {
525            if (!contractsLoaded) {
526                synchronized(this) {
527                    if (!contractsLoaded) {
528                        contracts   = new TreeMap();
529    
530                        try {
531                            Iterator it = dataManager.getContracts(symbol);
532                            while (it.hasNext()) {
533                                Contract contract = (Contract)it.next();
534                                addContract(contract);
535                            }
536                        } catch (IOException e) {
537                            e.printStackTrace();
538                        }
539                    }
540                    contractsLoaded = true;
541                }
542            }
543        }

545        /**
546         *  Format a price to the appropriate number of decimal places so that
547         *  the number of decimal places are fixed even if they end in zeros.
548         *
549         *  @param  price   the price to be formatted
550         *  @return         the formatted price
551         */
552        public String formatPrice(double price) {
553            DecimalFormat format = null;
554            double tickSize = getTickSize();
555    
556            if (tickSize >= 1) {
557                format = new DecimalFormat("0");
558            } else {
559                StringBuffer sb = new StringBuffer();
560                double value = tickSize;
561                do {
562                    sb.append("0");
563                    value *= 10;
564                } while (((value - (int)value) != 0.0) && (sb.length() < 3));
565                format = new DecimalFormat("0." + sb);
566            }
567    
568            return format.format(price);
569        }

650        /**
651         *  Load the Commodity table and store it to be keyed by symbol.
652         */
653        private static void loadTable() {
654            if (!loaded) {
655                synchronized(Commodity.class) {
656                    if (!loaded) {
657                        tableBySymbol   = new TreeMap();
658                        tableByName     = new TreeMap();
659    
660                        try {
661                            Iterator it = dataManager.getCommodities();
662                            while (it.hasNext()) {
663                                Commodity commodity = (Commodity)it.next();
664                                tableBySymbol.put(commodity.getSymbol(), commodity);
665                                tableByName.put(commodity.getNameExchange(), commodity);
666                            }
667                        } catch (IOException e) {
668                            e.printStackTrace();
669                        }
670                    }
671                    loaded = true;
672                }
673            }
674        }
