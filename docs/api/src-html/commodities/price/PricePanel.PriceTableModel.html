227            /**
228             *  Invoked when the date changes.
229             *  @param  e   the PriceEvent
230             */
231            public void changeDate(PriceEvent e) {
232                int row = contractData.indexOf(e.getContract());
233                String priceDate = FormatDate.formatDate(e.getDate(), DATE_FORMAT);
234                if ((getValueAt(row, DATE_COLUMN) == null) || (((String)getValueAt(row, DATE_COLUMN)).compareTo(priceDate) <= 0)) {
235                    setValueAt(priceDate, row, DATE_COLUMN);
236                    fireTableRowsUpdated(row, row);
237                }
238            }
239

242             *  @param  e   the PriceEvent

289        /**
290         *  The cell renderer for laying out the appearence of individual
291         *  JTable cells.
292         *
293         *  @author J.R. Titko
294         */
295        public class PriceTableCellRenderer extends DefaultTableCellRenderer {
296        /* *************************************** */
297        /* *** CONSTRUCTORS                    *** */
298        /* *************************************** */
299            /** Create a PriceTableCellRenderer. */
300            public PriceTableCellRenderer() {
301                setOpaque(true);
302            }
303    
304        /* **************************************** */
305        /* *** DefaultTableCellRenderer METHODS *** */
306        /* **************************************** */
307            /**
308             *  Gets the rendering of a cell based on information contained
309             *  in those cells.
310             *
311             *  @param  table       the JTable
312             *  @param  value       the value to assign to the cell at [row, column]
313             *  @param  isSelected  true if cell is selected
314             *  @param  hasFocus    if true, render cell appropriately.
315             *  @param  row         the row of the cell to render
316             *  @param  column      the column of the cell to render
317             *  @return             this component as a properly rendered cell
318             */
319            public Component getTableCellRendererComponent(JTable table, Object value, boolean isSelected, boolean hasFocus, int row, int column) {
320                super.getTableCellRendererComponent(table, value, isSelected, hasFocus, row, column);
321    
322                // Alignment
323                switch (column) {
324                    case CONTRACT_COLUMN:
325                            setHorizontalAlignment(LEFT);
326                            break;
327                    case DATE_COLUMN:
328                            setHorizontalAlignment(CENTER);
329                            break;
330                    case SHORT_TERM_COLUMN:
331                    case MEDIUM_TERM_COLUMN:
332                    case LONG_TERM_COLUMN:
333                            setHorizontalAlignment(CENTER);
334                            break;
335                    default:
336                            setHorizontalAlignment(RIGHT);
337                }
338    
339                // Font
340                switch (column) {
341                    case CONTRACT_COLUMN:
342                    case SHORT_TERM_COLUMN:
343                    case MEDIUM_TERM_COLUMN:
344                    case LONG_TERM_COLUMN:
345                            setFont(new Font(getFont().getName(), Font.BOLD, getFont().getSize()));
346                            break;
347                    default:
348                            setHorizontalAlignment(RIGHT);
349                }
350    
351                // Colors
352                switch (column) {
353                    case SHORT_TERM_COLUMN:
354                    case MEDIUM_TERM_COLUMN:
355                    case LONG_TERM_COLUMN:
356                            if (value != null) {
357                                String s = value.toString();
358    /*
359                                if (Recommendation.BUY_LABEL.equals(s) || Recommendation.SETTLE_BUY_LABEL.equals(s)) {
360                                    setBackground(Color.GREEN);
361                                    setForeground(Color.BLACK);
362                                } else if (Recommendation.SELL_LABEL.equals(s) || Recommendation.SETTLE_SELL_LABEL.equals(s)) {
363                                    setBackground(Color.CYAN);
364                                    setForeground(Color.BLACK);
365                                } else if (Recommendation.NO_ACTION_LABEL.equals(s)) {
366                                    setBackground(Color.WHITE);
367                                    setForeground(Color.BLACK);
368                                } else {
369                                    setBackground(Color.WHITE);
370                                    setForeground(Color.BLACK);
371                                }
372    */
373                            }
374                            break;
375                    default:
376                        if (isSelected || hasFocus) {
377                            setBackground(table.getSelectionBackground());
378                            setForeground(table.getSelectionForeground());
379                        } else {
380                            if (row == 0) {
381                                int month = Month.byAbbrev(((String)tableModel.getValueAt(0, CONTRACT_COLUMN)).substring(0,3)).getNumber();
382                                month -= 1;
383                                if (month <= (CommodityCalendar.getInstance().get(Calendar.MONTH) + 1)) {
384                                    setBackground(Color.RED);
385                                    setForeground(Color.BLACK);
386                                }
387                            }
388                            if ((column != CONTRACT_COLUMN) && ((String)tableModel.getValueAt(row, DATE_COLUMN)).length() > 0) {
389                                Prices prices = ((Contract)contractData.get(row)).getPrices();
390                                CommodityCalendar calendar = new CommodityCalendar(prices.getDate());
391    
392                                int month = Integer.parseInt(((String)tableModel.getValueAt(row, DATE_COLUMN)).substring(0,2));
393                                int day   = Integer.parseInt(((String)tableModel.getValueAt(row, DATE_COLUMN)).substring(3,5));
394                                if ((month != (CommodityCalendar.getInstance().get(Calendar.MONTH) + 1)) ||
395                                    (day   != (CommodityCalendar.getInstance().get(Calendar.DATE)))) {
396                                    setBackground(Color.YELLOW);
397                                    setForeground(Color.DARK_GRAY);
398                                }
399                            }
400                        }
401                    }
402                return this;
403            }
404        }

409        /**
410         *  Create a new PricePanel for the commodity with a reference
411         *  back to the PricingPanel that this panel will reside in.
412         *
413         *  @param  commodity       the commodity to set up in this PricePanel
414         */
415        public PricePanel (Commodity commodity) {
416            this.commodity = commodity;
417            this.myPricePanel = this;
418    
419            setupGUI();
420    
421            Iterator it = commodity.getContracts();
422            while (it.hasNext()) {
423                Contract contract = (Contract)it.next();
424                addRow(contract);
425            }
426            Contract.addSelectionListener(this);
427        }

442        /**
443         *  Set up the layout of the PricePanel with a header and JTable.
444         */
445        private void setupGUI() {
446            setLayout(new BorderLayout());
447            setBorder(BorderFactory.createLineBorder(Color.black));
448    
449            tableModel = new PriceTableModel();
450            table = new JTable(tableModel) {
451                public TableCellRenderer getCellRenderer(int row, int column) {
452                    return new PriceTableCellRenderer();
453                }
454            };
455            table.setSelectionBackground(Color.BLACK);
456            table.setSelectionForeground(Color.WHITE);
457            table.getColumn(tableModel.getColumnName(CONTRACT_COLUMN)).setPreferredWidth(100);
458            table.getColumn(tableModel.getColumnName(SHORT_TERM_COLUMN)).setPreferredWidth(32);
459            table.getColumn(tableModel.getColumnName(MEDIUM_TERM_COLUMN)).setPreferredWidth(32);
460            table.getColumn(tableModel.getColumnName(LONG_TERM_COLUMN)).setPreferredWidth(32);
461            JTableHeader tableHeader = table.getTableHeader();
462            tableHeader.setBackground(Color.BLUE);
463            tableHeader.setForeground(Color.WHITE);
464    
465    
466    
467            JPanel titlePanel = new JPanel();
468            titlePanel.setLayout(new BorderLayout());
469            titlePanel.setBackground(Color.GREEN);
470    
471            JLabel contPrice = new JLabel("  Cost=$" + commodity.getMargin());
472            contPrice.setFont(new Font(contPrice.getFont().getName(), Font.BOLD, contPrice.getFont().getSize()-1));
473            contPrice.setForeground(Color.BLACK);
474    
475            JLabel title = new JLabel(commodity.getName() + " (" + commodity.getExchange() + ") " +
476                                      commodity.getUnitSize() + " " + commodity.getUnitType() + "; " +
477                                      commodity.getUnitPrice() + " per " + commodity.getUnitType());
478            title.setHorizontalAlignment(JLabel.CENTER);
479            title.setForeground(Color.BLACK);
480    
481            JLabel price = new JLabel("TS=" + commodity.getTickSize() + " ($" + commodity.getTickPrice() + ")  ");
482            price.setFont(new Font(price.getFont().getName(), Font.BOLD, price.getFont().getSize()-1));
483            price.setForeground(Color.BLACK);
484    
485            titlePanel.add(contPrice, BorderLayout.WEST);
486            titlePanel.add(title, BorderLayout.CENTER);
487            titlePanel.add(price, BorderLayout.EAST);
488    
489            jsp = new JScrollPane(JScrollPane.VERTICAL_SCROLLBAR_ALWAYS, JScrollPane.HORIZONTAL_SCROLLBAR_AS_NEEDED);
490            jsp.setPreferredSize(new Dimension(PREFERRED_PANEL_WIDTH, PREFERRED_PANEL_HEIGHT));
491            jsp.setViewportView(table);
492    
493            add(BorderLayout.NORTH, titlePanel);
494            add(BorderLayout.CENTER, jsp);
495    
496            ListSelectionModel listSelectionModel = table.getSelectionModel();
497            listSelectionModel.addListSelectionListener(new ListSelectionListener() {
498                public void valueChanged(ListSelectionEvent e) {
499                    //Ignore extra messages.
500                    if (e.getValueIsAdjusting()) return;
501    
502                    ListSelectionModel lsm = (ListSelectionModel)e.getSource();
503                    if (lsm.isSelectionEmpty()) {
504                        //no rows are selected
505                    } else {
506                        ((Contract)contractData.get(lsm.getMinSelectionIndex())).select();
507                    }
508                }
509            });
510        }

525        /**
526         *  Invoked when the contract selection changes.
527         *  @param  e   the ContractSelectionEvent
528         */
529        public void selectContract(ContractSelectionEvent e) {
530            int newRow = contractData.indexOf((Contract)e.getSource());
531            if (newRow == -1) {
532                table.clearSelection();
533            } else if (!table.isRowSelected(newRow)) {
534                table.setRowSelectionInterval(newRow, newRow);
535            }
536        }
