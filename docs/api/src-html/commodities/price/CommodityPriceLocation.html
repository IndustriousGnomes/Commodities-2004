003    package commodities.price;
004    
005    import java.io.*;
006    import java.util.*;
007    
008    import commodities.commodity.*;
009    import commodities.dataaccess.*;
010    
011    /**
012     *  The PriceLocation class contains the location in which a commodity is located
013     *
014     *  @author J.R. Titko
015     *  @version 1.0
016     *  @update 2004.11.11
017     */
018    
019    public class CommodityPriceLocation {
020    /* *************************************** */
021    /* *** CLASS VARIABLES                 *** */
022    /* *************************************** */
023        /** A reference to the data manager. */
024        private static DataManagerInterface dataManager = DataManagerFactory.instance();
025        /** Table of locations by page. */
026        private static Map tableByPage;
027    
028    /* *************************************** */
029    /* *** INSTANCE VARIABLES              *** */
030    /* *************************************** */
031        /** The page number the commodity is located on. */
032        private int     page;
033        /** The location within a page the commodity is located on. */
034        private int     location;
035        /** The commodity symbol. */
036        private String  symbol;
037        /** The title of the tab */
038        private String  title;
039    
040    /* *************************************** */
041    /* *** CONSTRUCTORS                    *** */
042    /* *************************************** */
043        /**
044         *  Creates a month with the different representations for the month.
045         *
046         *  @param  page        The page number the commodity is located on
047         *  @param  location    The location within a page the commodity is located on
048         *  @param  symbol      The commodity symbol
049         */
050        public CommodityPriceLocation(int page, int location, String symbol, String title) {
051            this.page       = page;
052            this.location   = location;
053            this.symbol     = symbol.trim();
054            this.title      = title.trim();
055        }
056    
057    /* *************************************** */
058    /* *** GET AND SET METHODS             *** */
059    /* *************************************** */
060        /**
061         *  Get the page number.
062         *  @return     the page
063         */
064        public int getPage() {
065            return page;
066        }
067    
068        /**
069         *  Get the location.
070         *  @return     the location
071         */
072        public int getLocation() {
073            return location;
074        }
075    
076        /**
077         *  Get the commodity symbol.
078         *  @return     the symbol
079         */
080        public String getSymbol() {
081            return symbol;
082        }
083    
084        /**
085         *  Get the tab title.
086         *  @return     the title
087         */
088        public String getTitle() {
089            return title;
090        }
091    
092        /**
093         *  Set the location.
094         *  @param  location    the location
095         */
096        public void setLocation(int location) {
097            this.location = location;
098        }
099    
100    /* *************************************** */
101    /* *** STATIC METHODS                  *** */
102    /* *************************************** */
103        /**
104         *  Get the collection of page numbers.
105         *  @return     collection of pages
106         */
107        public static Collection pages() {
108            if (tableByPage == null) {
109                loadTable();
110            }
111            return tableByPage.keySet();
112        }
113    
114        /**
115         *  Get an iterator of contracts on a page.
116         *
117         *  @param  pg  page number
118         *  @return     collection of contracts names
119         */
120        public static Collection byPage(int pg) {
121            if (tableByPage == null) {
122                loadTable();
123            }
124            return (LinkedList)tableByPage.get(new Integer(pg));
125        }
126    
127        /**
128         *  The title of the tab.
129         *
130         *  @param  pg  page number
131         *  @return     the title
132         */
133        public static String getTitle(int pg) {
134            if (tableByPage == null) {
135                loadTable();
136            }
137            return ((CommodityPriceLocation)((LinkedList)tableByPage.get(new Integer(pg))).get(0)).getTitle();
138        }
139    
140        /**
141         *  Add a price location.
142         *
143         *  @param  cpl commodity price location
144         */
145        public static void addPriceLocation(CommodityPriceLocation cpl) {
146            try {
147                dataManager.putCommodityPageLocation(cpl);
148                LinkedList locs = (LinkedList)tableByPage.get(new Integer(cpl.getPage()));
149                if (locs == null) {
150                    locs = new LinkedList();
151                    tableByPage.put(new Integer(cpl.getPage()), locs);
152                }
153    
154                locs.add(cpl);
155            } catch (IOException e) {
156                e.printStackTrace();
157            }
158        }
159    
160        /**
161         *  Remove one or more price location from a page identified by commodity name and exchange.
162         *
163         *  @param  page    page to remove commodities from
164         *  @param  name    name of the commodity
165         */
166        public static void removePriceLocations(int page, String name[]) {
167            try {
168                LinkedList toRemove = new LinkedList();
169                LinkedList locs = (LinkedList)tableByPage.get(new Integer(page));
170                Iterator it = locs.iterator();
171                while (it.hasNext()) {
172                    CommodityPriceLocation cpl = (CommodityPriceLocation)it.next();
173                    for (int i = 0; i < name.length; i++) {
174                        if (cpl.getSymbol().equals(Commodity.byNameExchange(name[i]).getSymbol())) {
175                            toRemove.add(cpl);
176                            break;
177                        }
178                    }
179                }
180                locs.removeAll(toRemove);
181                dataManager.removeCommodityPriceLocations(page, locs);
182    
183            } catch (IOException e) {
184                e.printStackTrace();
185            }
186        }
187    
188        /**
189         *  Load the Commodity Price Location table and store it to be keyed by
190         *  page and location.
191         */
192        private static void loadTable() {
193            tableByPage = new TreeMap();
194            LinkedList locs = new LinkedList();
195    
196            try {
197                Iterator it = dataManager.getCommodityPageLoc();
198                int pg = 0;
199                tableByPage.put(new Integer(pg), locs);
200                while (it.hasNext()) {
201                    CommodityPriceLocation cpl = (CommodityPriceLocation)it.next();
202                    if (cpl.getPage() != pg) {
203                        pg = cpl.getPage();
204                        locs = new LinkedList();
205                        tableByPage.put(new Integer(pg), locs);
206                    }
207                    locs.add(cpl);
208                }
209            } catch (IOException e) {
210                e.printStackTrace();
211            }
212        }
