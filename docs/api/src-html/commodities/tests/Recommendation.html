003    package commodities.tests;
004    
005    /**
006     *  Recommendation contains a test's recommendation on what action
007     *  to take with a commodity (buy/sell).
008     *
009     *  @author J.R. Titko
010     *  @version 1.0
011     *  @update 2004.11.11
012     */
013    
014    public class Recommendation implements Comparable {
015    /* *************************************** */
016    /* *** CLASS VARIABLES                 *** */
017    /* *************************************** */
018        /** Value for recommending a sell action */
019        public static final int SELL       = 2;
020        /** Value for recommending a sell action if a contract is currently bought */
021        public static final int SETTLE_SELL   = 1;
022        /** Value for recommending no action to be taken */
023        public static final int NO_ACTION  = 0;
024        /** Value for recommending a buy action if a contract is currently sold */
025        public static final int SETTLE_BUY    = -1;
026        /** Value for recommending a buy action */
027        public static final int BUY        = -2;
028    
029        /** Label for recommending a sell action */
030        public static final String  SELL_LABEL          = "S";
031        /** Label for recommending a sell action if a contract is currently bought */
032        public static final String  SETTLE_SELL_LABEL   = "SS";
033        /** Label for recommending no action to be taken */
034        public static final String  NO_ACTION_LABEL     = "-";
035        /** Label for recommending a buy action if a contract is currently sold */
036        public static final String  SETTLE_BUY_LABEL    = "SB";
037        /** Label for recommending a buy action */
038        public static final String  BUY_LABEL           = "B";
039    
040        /** Recommendation label names */
041        private static final String labels[] = {BUY_LABEL, SETTLE_BUY_LABEL, NO_ACTION_LABEL, SETTLE_SELL_LABEL, SELL_LABEL};
042    
043    
044    /* *************************************** */
045    /* *** INSTANCE VARIABLES              *** */
046    /* *************************************** */
047        /** Name of the test that this recommendation is for */
048        private String testName;
049        /** The buy/sell recommendation of this test */
050        private int buySell;
051        /** The recommended stoploss. */
052        private double stoploss;
053    
054    /* *************************************** */
055    /* *** CONSTRUCTORS                    *** */
056    /* *************************************** */
057        /**
058         *  Create a recommendation for the given test.
059         *
060         *  @param  testName        the name of the test to use in displays
061         *  @param  buySell         the buy/sell recommendation for this test
062         */
063        public Recommendation(String testName, int buySell) {
064            this(testName, buySell, 0.0);
065        }
066    
067        /**
068         *  Create a recommendation for the given test with a stoploss amount.
069         *  The stoploss amount assumes a preferred loss amount of $0.  The
070         *  actual stoploss amount should be adjusted accordingly.  Even if the
071         *  recommendation is no action, the stoploss should still indicate an
072         *  appropriate exit price.
073         *
074         *  @param  testName        the name of the test to use in displays
075         *  @param  buySell         the buy/sell recommendation for this test
076         *  @param  stoploss        the stoploss recommendation
077         */
078        public Recommendation(String testName, int buySell, double stoploss) {
079            this.testName = testName;
080            this.buySell = buySell;
081            this.stoploss = stoploss;
082        }
083    
084    /* *************************************** */
085    /* *** GET AND SET METHODS             *** */
086    /* *************************************** */
087        /**
088         *  Get the test name.
089         *  @return     name of the test that this recommendation is for
090         */
091        public String getTestName() {
092            return testName;
093        }
094    
095        /**
096         *  Get the recommendation.
097         *  @return     the recommendation
098         */
099        public int getBuySellRecommendation() {
100            return buySell;
101        }
102    
103        /**
104         *  Get the label of the buy/sell recommendation.
105         *
106         *  @return         the label used for the recommendation
107         */
108        public String getBuySellRecommendLabel() {
109            return labels[buySell + 2];
110        }
111    
112        /**
113         *  Get the recommended stoploss price.  This amount takes into consideration a
114         *  preferred stoploss cost of $0.  The final stoploss amount should be adjusted
115         *  accordingly.
116         *
117         *  @return     the stoploss amount
118         */
119        public double getStoploss() {
120            return stoploss;
121        }
122    
123    /* *************************************** */
124    /* *** Comparable METHODS              *** */
125    /* *************************************** */
126        /**
127         *  Compare the objects for sorting.
128         *  @param  o2      the object to compare to
129         *  @return         negative, zero, possitive number depending on
130         *                  this object to the passed object
131         */
132        public int compareTo(Object o2) {
133            return testName.compareTo(((Recommendation)o2).getTestName());
134        }
135    
136    }
