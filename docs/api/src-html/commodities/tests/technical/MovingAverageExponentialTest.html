232        /**
233         *  This method loads the data used for each moving average test that was
234         *  determined to be an optimal moving average test for a contract.
235         */
236        protected void loadData() {
237            Map     priceList = contract.getPriceList();
238            Prices  prices = null;
239            java.util.Date  statsDate = null;
240    
241    // For each day's prices...
242            Iterator it = priceList.keySet().iterator();
243            while (it.hasNext()) {
244                statsDate = (java.util.Date)it.next();
245                prices = (Prices)priceList.get(statsDate);
246    
247    // recalculate the statististics data by adding it to the queue for that moving average.
248                Iterator it2 = statCollection.iterator();
249                while (it2.hasNext()) {
250                    ((Stats)it2.next()).addStatData(statsDate, prices);
251                }
252            }
253            it = statCollection.iterator();
254            while (it.hasNext()) {
255                Stats s = (Stats)it.next();
256                Iterator it2 = s.getData().iterator();
257                while (it2.hasNext()) {
258                    double value = ((Double)it2.next()).doubleValue();
259                    if (value < statDataMin) {
260                        statDataMin = value;
261                    }
262                    if (value > statDataMax) {
263                        statDataMax = value;
264                    }
265                }
266            }
267        }

269        /**
270         *  Retrieves the parameter data for moving averages
271         */
272        protected void loadParameters() {
273            DBConnectionPool dbcp = null;
274    
275            try {
276                dbcp = DBConnectionPool.instance();
277            } catch (SQLException e) {
278                throw new RuntimeException("Database failure");
279            }
280    
281            Connection connect = null;
282            try {
283                connect = dbcp.retrieveConnection();
284    
285                StringBuffer sb = new StringBuffer();
286                sb.append("select id, param_percent " +
287                          "from " + sqlTable + " " +
288                          "where commodity_id = '" + contract.getSymbol() + "' " +
289                          "  and contract_month = '" + contract.getMonth() + "' ");
290                String sqlTxt = sb.toString();
291    
292                Statement stmt = connect.createStatement();
293                ResultSet rs = stmt.executeQuery(sqlTxt);
294                while (rs.next()) {
295                    int id = rs.getInt(1);
296                    int paramPercent = rs.getInt(2);
297                    Stats stats = new Stats(paramPercent);
298                    statCollection.add(stats);
299                    tableByTestId.put(new Integer(id), stats);
300                }
301                rs.close();
302                stmt.close();
303            } catch (SQLException e) {
304                e.printStackTrace();
305            } finally {
306                if (connect != null) {
307                    dbcp.returnConnection(connect);
308                }
309            }
310        }

461            Stats stats = (Stats)statsAbs;

465                return Recommendation.NO_ACTION;
466            }
467    
468            int keyIndex = statsKeys.indexOf(date);
469            if (firstDay && (keyIndex < 2)) {

475            double y2 = ((Double)stats.getData(date)).doubleValue();

477            double slope = y2 - y1;
478    
479            if ((slope > 0.0) && (priceClose >= y2)) {                      // Positive slope and price above moving average

481            } else if ((slope < 0.0) && (priceClose <= y2)) {               // Negative slope and price below moving average
