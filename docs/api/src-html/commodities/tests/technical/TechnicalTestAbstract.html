100            /**
101             *  Gets the number of pixels that should be added as white
102             *  space to the top and bottom of the graph when calculating
103             *  the scale of the graph.  This value is inside the graph's
104             *  frame and therefore reduces the amount of drawing space
105             *  by twice its amount.

143            /**
144             *  Draws the title of the graph at the top of the graph offset by
145             *  the other graph titles that are alread there.
146             *
147             *  @return         always returns true
148             */
149            public boolean redrawTitle(Graphics g, int yOffset) {
150                Rectangle graphRect = graph.getGraphRect();
151                int xAxisLeft       = (int)graphRect.getX();
152    
153                boolean firstTest = true;
154                int graphSeq = 0;
155    
156                g.setColor(Color.BLACK);
157                String testNameString = testName + ": ";
158                g.drawString(testNameString, xAxisLeft, yOffset+g.getFontMetrics().getAscent());
159                int testNameOffset = g.getFontMetrics().stringWidth(testNameString);
160    
161                Iterator it = statCollection.iterator();
162    //            Iterator it = recommendCollection.iterator();
163                while (it.hasNext()) {
164                    if (!firstTest) {
165                        g.setColor(colors[graphNbr][0]);
166                        g.drawString(", ", xAxisLeft + testNameOffset, yOffset+g.getFontMetrics().getAscent());
167                        testNameOffset += g.getFontMetrics().stringWidth(", ");
168                    }
169                    g.setColor(colors[graphNbr][graphSeq]);
170                    firstTest = false;
171                    String title = ((StatsAbstract)it.next()).getName();
172                    g.drawString(title, xAxisLeft + testNameOffset, yOffset+g.getFontMetrics().getAscent());
173                    testNameOffset += g.getFontMetrics().stringWidth(title);
174    
175                    graphSeq++;
176                    if (graphSeq >= colors[graphNbr].length) {
177                        graphSeq--;
178                    }
179                }
180                return true;
181            }

183            /**
184             *  Sets up the drawing space, scale, and then draws the graph.
185             *
186             *  @param  e   the event containing the graph to draw into
187             */
188            public void redrawComponent(GraphRedrawEvent e) {
189                Graphics g = e.getGraphics();
190    
191                Rectangle graphRect    = graph.getGraphRect();
192                int xAxisLeft   = (int)graphRect.getX();
193                int xAxisRight  = (int)graphRect.getX() + (int)graphRect.getWidth();
194                int yAxisBottom = (int)graphRect.getY() + (int)graphRect.getHeight();
195    
196                double minValue = e.getMinValue();
197                int    xScale   = e.getXScale();
198                double yScale   = e.getYScale() * e.getMinTickSize();
199    
200                int graphSeq = 0;
201                // Draw the price ranges
202                Iterator it = statCollection.iterator();
203    //            Iterator it = recommendCollection.iterator();
204                while (it.hasNext()) {
205                    StatsAbstract stats = (StatsAbstract)it.next();
206                    g.setColor(colors[graphNbr][graphSeq]);
207    
208                    redrawGraph(g, stats, xAxisLeft, xAxisRight, yAxisBottom, minValue, xScale, yScale);
209    
210                    graphSeq++;
211                    if (graphSeq >= colors[graphNbr].length) {
212                        graphSeq--;
213                    }
214                }
215            }

218             *  Aligns the right hand side of the graph to the next monday and then draws the graph.
219             *
220             *  @param  g           the graphics area to draw in
221             *  @param  stats       the statistics to be draw
222             *  @param  xAxisLeft   the possision of the left side of the graph
223             *  @param  xAxisRight  the possision of the right side of the graph
224             *  @param  yAxisBottom the possision of the bottom side of the graph
225             *  @param  minValue    the minimum value to be graphed
226             *  @param  xScale      the horizontal scale the graph is to be drawn in
227             *  @param  yScale      the vertical scale the graph is to be drawn in
228             */
229            public void redrawGraph(Graphics      g,
230                                    StatsAbstract stats,
231                                    int           xAxisLeft,
232                                    int           xAxisRight,
233                                    int           yAxisBottom,
234                                    double        minValue,
235                                    int           xScale,
236                                    double        yScale) {
237    
238                // Determine the date of the right side of the graph.
239                CommodityCalendar calendar = (CommodityCalendar)CommodityCalendar.getInstance();
240                calendar.clearTime();
241                calendar.setNextDayOfWeek(CommodityCalendar.MONDAY);
242                drawGraph(g, stats, calendar, xAxisLeft, xAxisRight, yAxisBottom, minValue, xScale, yScale);
243            }

249             *  @param  stats       the statistics to be drawn
250             *  @param  calendar    the calendar to be used to process the graph starting on a monday

291            /**
292             *  Draw the a line graph.
293             *
294             *  @param  g           the graphics area to draw in
295             *  @param  stats       the statistics to be drawn
296             *  @param  calendar    the calendar to be used to process the graph starting on a monday
297             *  @param  xAxisLeft   the position of the left side of the graph
298             *  @param  xAxisRight  the position of the right side of the graph
299             *  @param  yAxisBottom the position of the bottom side of the graph
300             *  @param  minValue    the minimum value to be graphed
301             *  @param  xScale      the horizontal scale the graph is to be drawn in
302             *  @param  yScale      the vertical scale the graph is to be drawn in
303             */
304            public void drawGraph(Graphics      g,
305                                  StatsAbstract stats,
306                                  CommodityCalendar calendar,
307                                  int           xAxisLeft,
308                                  int           xAxisRight,
309                                  int           yAxisBottom,
310                                  double        minValue,
311                                  int           xScale,
312                                  double        yScale) {
313                Color  graphColor = g.getColor();
314                double oldValue = -1.0;
315                double newValue = -1.0;
316                int    xOld = xAxisRight;
317                for (int x = xAxisRight; x > xAxisLeft; x -= xScale) {
318                    Double value = (Double)stats.getData(calendar.getTime());
319                    if (value != null) {
320                        newValue = value.doubleValue();
321                        if (oldValue!= -1.0) {
322                            g.setColor(graphColor);
323                            g.drawLine(xOld, (int)(yAxisBottom - (oldValue - minValue) / yScale), x, (int)(yAxisBottom - (newValue - minValue) / yScale));
324                            g.drawLine(xOld, (int)(yAxisBottom + 1 - (oldValue - minValue) / yScale), x, (int)(yAxisBottom + 1 - (newValue - minValue) / yScale));
325    
326                            int recommend = makeBuySellRecommendation(calendar.getTime(), stats);
327                            if (recommend > 0) {
328                                g.setColor(Color.WHITE);
329                                g.fillOval(x - 5, (int)(yAxisBottom - (newValue - minValue) / yScale - 30) - 5,  9, 9);
330                                g.setColor(Color.BLACK);
331                                g.drawOval(x - 5, (int)(yAxisBottom - (newValue - minValue) / yScale - 30) - 5, 10, 10);
332    
333                                g.setColor(graphColor);
334                                Polygon poly = new SellArrow();
335                                poly.translate(x, (int)(yAxisBottom - (newValue - minValue) / yScale - 30));
336                                g.fillPolygon(poly);
337                            } else if (recommend < 0) {
338                                g.setColor(Color.WHITE);
339                                g.fillOval(x - 5, (int)(yAxisBottom - (newValue - minValue) / yScale + 30) - 5,  9, 9);
340                                g.setColor(Color.BLACK);
341                                g.drawOval(x - 5, (int)(yAxisBottom - (newValue - minValue) / yScale + 30) - 5, 10,10);
342    
343                               g.setColor(graphColor);
344                                Polygon poly = new BuyArrow();
345                                poly.translate(x, (int)(yAxisBottom - (newValue - minValue) / yScale + 30));
346                                g.fillPolygon(poly);
347                            }
348                        }
349                        oldValue = newValue;
350                        xOld = x;
351                    }
352                    calendar.addWeekDays(-1);
353                }
354            }
355        }

785        /**
786         *  Remove extra statistics that are too close to another that is more optimial.
787         *  This prevents multiple statistics that are clumped together from all being
788         *  reported.
789         *
790         *  @param  tableByPercent  table of success percentages
791         *  @param  statsArray      array of success statistics
792         */
793        protected void removeExtraOptimizedStats(TreeMap tableByPercent,
794                                                 StatsAbstract statsArray[]) {
795            Iterator it = tableByPercent.keySet().iterator();
796            while (it.hasNext()) {
797                int term = ((Integer)tableByPercent.get(it.next())).intValue();
798                int validTestSpan = 0;
799                if (term <= SHORT_TERM) {
800                    validTestSpan = 3;
801                } else if (term <= MEDIUM_TERM) {
802                    validTestSpan = 4;
803                } else {            // LONG_TERM
804                    validTestSpan = 6;
805                }
806                for (int i = term - validTestSpan; i <= (term + validTestSpan); i++) {
807                    if ((i > 0) && (i != term) && (i < statsArray.length) && (statsArray[i] != null)) {
808                        statsArray[term] = null;
809                        break;
810                    }
811                }
812            }
813        }
