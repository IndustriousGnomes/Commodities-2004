009    /**
010     *  MovingAverageTriangle calculates the moving average on a set of data.
011     *  The sourceMap that is passed to MovingAverageTriangle must be of
012     *  date/Double values in ascending order by date.  MovingAverageTriangle
013     *  will then traverse the sourceMap to determine the moving average
014     *  for a given date or for the entire range.
015     *
016     *  @author J.R. Titko
017     */
018    
019    public class MovingAverageTriangle extends MovingAverage {
020    /* *************************************** */
021    /* *** INSTANCE VARIABLES              *** */
022    /* *************************************** */
023        /** The smoothing Moving Average interval */
024        private int smoothedInterval = 0;
025        /** The window of smoothed data being analyzed for this interval. */
026        private LinkedList smoothedWindow = new LinkedList();
027        /** The value of the smoothed interval within the window. */
028        private double smoothedWindowTotal = 0;
029    
030    
031    /* *************************************** */
032    /* *** CONSTRUCTORS                    *** */
033    /* *************************************** */
034        public MovingAverageTriangle(int interval, TreeMap sourceMap) {
035            this.interval = interval;
036            this.sourceMap = sourceMap;
037            smoothedInterval = (int)(interval / 2.0 + 0.5);
038    
039            calculateMovingAverages();
040        }
041    
042    /* *************************************** */
043    /* *** INSTANCE METHODS                *** */
044    /* *************************************** */
045        /**
046         *  Calculate the moving averages.
047         */
048        protected void calculateMovingAverages() {
049            smoothedWindow = new LinkedList();
050            smoothedWindowTotal = 0;
051            super.calculateMovingAverages();
052        }
053    
054        /**
055         *  Calculate the moving averages.
056         */
057        protected void calculateMovingAverage(java.util.Date date, Double value) {
058            window.addLast(value);
059            windowTotal += (value).doubleValue();
060    
061            if (window.size() > interval) {
062                windowTotal -= ((Double)window.removeFirst()).doubleValue();
063    
064                double primaryValue = ((windowTotal / interval) * 1000) / 1000;
065    
066                smoothedWindow.addLast(new Double(primaryValue));
067                smoothedWindowTotal += primaryValue;
068                if (smoothedWindow.size() > smoothedInterval) {
069                    smoothedWindowTotal -= ((Double)smoothedWindow.removeFirst()).doubleValue();
070    
071                    double smoothedValue = ((smoothedWindowTotal / smoothedInterval) * 1000) / 1000;
072                    dataMap.put(date, new Double(smoothedValue));
073                }
074            }
075        }
076    }
