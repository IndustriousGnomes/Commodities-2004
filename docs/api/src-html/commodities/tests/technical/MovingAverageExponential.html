003    package commodities.tests.technical;
004    
005    import java.util.*;
006    
007    /**
008     *  MovingAverageExponential calculates the moving average on a set of data.
009     *  The sourceMap that is passed to MovingAverageExponential must be of
010     *  date/Double values in ascending order by date.  MovingAverageExponential
011     *  will then traverse the sourceMap to determine the moving average
012     *  for a given date or for the entire range.
013     *
014     *  @author J.R. Titko
015     *  @version 1.0
016     *  @update 2004.11.12
017     */
018    
019    public class MovingAverageExponential extends MovingAverage {
020    /* *************************************** */
021    /* *** INSTANCE VARIABLES              *** */
022    /* *************************************** */
023        /** The value of the current moving average. */
024        private double movingAverage = 0;
025    
026    /* *************************************** */
027    /* *** CONSTRUCTORS                    *** */
028    /* *************************************** */
029        public MovingAverageExponential(int expPercent, TreeMap sourceMap) {
030            this.interval = expPercent;
031            this.sourceMap = sourceMap;
032            calculateMovingAverages();
033        }
034    
035    /* *************************************** */
036    /* *** INSTANCE METHODS                *** */
037    /* *************************************** */
038        /**
039         *  Calculate the moving averages.
040         */
041        protected void calculateMovingAverages() {
042            movingAverage = 0;
043            super.calculateMovingAverages();
044        }
045    
046        /**
047         *  Calculate the moving averages.
048         */
049        protected void calculateMovingAverage(java.util.Date date, Double value) {
050            if (movingAverage > 0) {
051                movingAverage = value.doubleValue() * (interval / 100.0) + movingAverage * ((100 - interval) / 100.0);
052            } else {
053                movingAverage = value.doubleValue();
054            }
055            dataMap.put(date, new Double(movingAverage));
056        }
057    }
