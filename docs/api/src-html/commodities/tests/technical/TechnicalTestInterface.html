003    package commodities.tests.technical;
004    
005    import commodities.tests.*;
006    
007    /**
008     *  The TechnicalTestInterface is a standard interface for all of the
009     *  technical tests.
010     *
011     *  All classes that extend this interface must contain the following 2 constructors:
012     *      Test() - Used for retrieving the name of the test ONLY!
013     *      Test(commodities.dataaccess.Contract) - For all other accesses.
014     *  This is required since the getName() method must be included in every class but
015     *  is a static name that must also be used to populate the Test menu through reflection.
016     *
017     *  @author J.R. Titko
018     *  @version 1.0
019     *  @update 2004.11.11
020     */
021    
022    public interface TechnicalTestInterface {
023        /** Short Term days */
024        public static final int SHORT_TERM  = 15;
025        /** Medium Term days */
026        public static final int MEDIUM_TERM = 45;
027        /** Long Term days */
028        public static final int LONG_TERM   = 120;
029        /** Minumum success ratio. 40% allows good tests that might have had some minor mishaps but still profitable. */
030        public static final double MIN_SUCCESS_RATIO   = 0.40;
031    
032    
033        /**
034         *  Retrieve the name of the test.  This name will be the english style name
035         *  that will appear in the menu and as a header for the graph.
036         */
037        public String getName();
038    
039        /**
040         *  This method will depict the results of the test.  Although this will
041         *  typically be in the form of a line graph, the graph can appear in
042         *  any appropriate format.
043         *
044         *  The location to draw the graph in will depend on each test.  If the
045         *  graph should be overlayed on the price graph, the test will have to
046         *  register with the price graph.  If it should be put in its own space,
047         *  then the statistical graph manager will have to be used.
048         */
049        public void graphResults();
050    
051        /**
052         *  Stop graphing this test.
053         */
054        public void stopGraphResults();
055    
056        /**
057         *  Perform an optimization of the test to determine the most profitable
058         *  combination(s) of parameters to use with the test.
059         */
060        public void optimizeTest();
061    
062        /**
063         *  Retrieve the recommendation resulting from the test for the
064         *  specified date.
065         *
066         *  @param  testId  the id of the test in the test table
067         *  @param  date    the date to get the recommendation for
068         *  @return         the recommendation
069         */
070        public Recommendation getRecommendation(int testId, java.util.Date date);
071    
072        /**
073         *  Make a recommendation based on the stats data.
074         *
075         *  @param  date        the date the recommendation is for
076         *  @param  statsObject the statistics as an object
077         *  @return             a recommendation based on this test
078         */
079        public Recommendation makeRecommendation(java.util.Date date, StatsAbstract stats);
080    
081        /**
082         *  Make a buy/sell recommendation based on the stats data.
083         *
084         *  @param  date        the date the recommendation is for
085         *  @param  statsObject the statistics as an object
086         *  @return             a buy/sell recommendation based on this test
087         */
088        public int makeBuySellRecommendation(java.util.Date date, StatsAbstract stats);
089    
090        /**
091         *  Make a stop loss recommendation based on the stats data.
092         *
093         *  @param  date        the date the recommendation is for
094         *  @param  statsObject the statistics as an object
095         *  @return             a stop loss recommendation based on this test
096         */
097        public double makeStopLossRecommendation(java.util.Date date, StatsAbstract stats);
098    }
