160             *  Calculate the statistical data.
161             */
162            public void calculateStatData() {
163                TreeMap kMap = new TreeMap();
164                LinkedList window = new LinkedList();
165                double windowHigh = 0.0;
166                double windowLow = 9999.0;
167    
168                dataList = null;
169                keyList = null;
170    System.out.println("stats.calculateStatData() - sourceMap.size = " + sourceMap.size());
171                Iterator it = sourceMap.keySet().iterator();
172                while (it.hasNext()) {
173                    java.util.Date date = (java.util.Date)it.next();
174                    Prices prices = (Prices)sourceMap.get(date);
175    
176                    window.addLast(prices);
177    
178                    if (window.size() > kPeriods) {
179                        Prices oldPrices = (Prices)window.removeFirst();
180    
181                        if (prices.getHigh() > windowHigh) {
182                            windowHigh = prices.getHigh();
183                        }
184                        if (prices.getLow() < windowLow) {
185                            windowLow = prices.getLow();
186                        }
187    
188                        if  (((oldPrices.getHigh() == windowHigh) && (prices.getHigh() != windowHigh)) ||
189                             ((oldPrices.getLow() == windowLow) && (prices.getLow() != windowLow))) {
190    
191                            windowHigh = 0.0;
192                            windowLow = 9999.0;
193                            Iterator it2 = window.iterator();
194                            while (it2.hasNext()) {
195                                Prices p = (Prices)it2.next();
196                                if (p.getHigh() > windowHigh) {
197                                    windowHigh = p.getHigh();
198                                }
199                                if (p.getLow() < windowLow) {
200                                    windowLow = p.getLow();
201                                }
202                            }
203                        }
204    
205                        if (windowHigh != windowLow) {
206                            double k = ((prices.getClose() - windowLow) / (windowHigh - windowLow)) * 100;
207                            kMap.put(date, new Double(k));
208                        }
209                    }
210                }
211    
212                dMovingAverage = new MovingAverage(dPeriods, kMap);
213                Map dMap = dMovingAverage.getMap();
214    
215                it = dMap.keySet().iterator();
216                while (it.hasNext()) {
217                    java.util.Date dDate = (java.util.Date)it.next();
218                    dataMap.put(dDate, new StatData(((Double)kMap.get(dDate)).doubleValue(), ((Double)dMap.get(dDate)).doubleValue()));
219                }
220            }

585         *  Retrieves the parameter data for moving averages
586         */
587        protected void loadParameters() {
588            DBConnectionPool dbcp = null;
589    
590            try {
591                dbcp = DBConnectionPool.instance();
592            } catch (SQLException e) {
593                throw new RuntimeException("Database failure");
594            }
595    
596            Connection connect = null;
597            try {
598                connect = dbcp.retrieveConnection();
599    
600                StringBuffer sb = new StringBuffer();
601                sb.append("select id, k_periods, k_slowing_periods, d_periods " +
602                          "from " + sqlTable + " " +
603                          "where commodity_id = '" + contract.getSymbol() + "' " +
604                          "  and contract_month = '" + contract.getMonth() + "' ");
605                String sqlTxt = sb.toString();
606    
607                Statement stmt = connect.createStatement();
608                ResultSet rs = stmt.executeQuery(sqlTxt);
609                while (rs.next()) {
610                    int id = rs.getInt(1);
611                    Stats stats = new Stats(rs.getInt(2), rs.getInt(3), rs.getInt(4));
612                    statCollection.add(stats);
613                    tableByTestId.put(new Integer(id), stats);
614                }
615                rs.close();
616                stmt.close();
617            } catch (SQLException e) {
618                e.printStackTrace();
619            } finally {
620                if (connect != null) {
621                    dbcp.returnConnection(connect);
622                }
623            }
624        }

766        /**
767         *  Make a recommendation based on the stats data.
768         *
769         *  @param  date    the date the recommendation is for
770         *  @param  stats   the statistics
771         *  @return         a recommendation based on this test
772         */
773        public Recommendation makeRecommendation(java.util.Date date, StatsAbstract statsAbs) {
774            int recommendation = Recommendation.NO_ACTION;
775            Stats stats = (Stats)statsAbs;
776    
777            LinkedList statsKeys = stats.getKeys();
778            if (statsKeys.size() < 2) {
779    //System.out.println("Stochastic recommend NO ACTION (keys)");
780                return new Recommendation(testName, Recommendation.NO_ACTION);
781            }
782    
783            int keyIndex = statsKeys.indexOf(date);
784            if (keyIndex < 2) {
785    //System.out.println("Stochastic recommend NO ACTION (index)");
786                return new Recommendation(testName, Recommendation.NO_ACTION);
787            }
788    
789            java.util.Date yesterDate = (java.util.Date)statsKeys.get(keyIndex - 1);
790    
791            double k = ((StatData)stats.getData(date)).getK();
792            double kYesterday = ((StatData)stats.getData(yesterDate)).getK();
793            double d = ((StatData)stats.getData(date)).getD();
794            double dYesterday = ((StatData)stats.getData(yesterDate)).getD();
795    
796            if ((kYesterday < dYesterday) && (k > d)) {
797                recommendation = Recommendation.BUY;
798            } else if ((kYesterday > dYesterday) && (k < d)) {
799                recommendation = Recommendation.SELL;
800            }
801            return new Recommendation(testName, recommendation);
802        }
